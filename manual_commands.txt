Build knowledgebase:

python -c "
from src.knowledge_base import RAGFactory, RAGConfig, RAGBackend
from pathlib import Path

# Configure RAG
config = RAGConfig(
    backend=RAGBackend.MANUAL,
    vector_store='chroma',
    persist_dir='./data/chroma_db',
    collection_name='medical_knowledge',
)

# Create factory
factory = RAGFactory(config)

# Build knowledge base
print('Building knowledge base...')
stats = factory.build_knowledge_base(
    Path('./medical_knowledge'),
    clear_existing=True,
)

print(f'Knowledge base built!')
print(f'Stats: {stats}')
"

from src.knowledge_base import RAGFactory, RAGConfig, RAGBackend

Generated samples inspection in python:

python -c "
import json
from pathlib import Path

# Find the latest output file
output_dir = Path('./data/synthetic_output')
files = list(output_dir.glob('synthetic_data_*.jsonl'))
if not files:
    print('No output files found')
    exit()

latest_file = max(files, key=lambda p: p.stat().st_mtime)
print(f'Reading: {latest_file}')

# Load and display first sample
with open(latest_file) as f:
    sample = json.loads(f.readline())

print()
print('=== SAMPLE ===')
print(f'ID: {sample[\"id\"]}')
print()
print('--- Dialogue ---')
for turn in sample['dialogue'][:4]:
    print(f\"{turn['speaker']}: {turn['text'][:100]}...\")
print('...')
print()
print('--- Summary ---')
print(f\"Chief Complaint: {sample['summary']['chief_complaint']}\")
print(f\"Assessment: {sample['summary']['assessment'][:200]}...\")
print(f\"Plan: {sample['summary']['plan'][:200]}...\")
"
